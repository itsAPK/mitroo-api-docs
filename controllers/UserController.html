import {
  Body,
  Controller,
  Param,
  ParseIntPipe,
  Post,
  Put,
  Get,
  Inject,
  UseGuards,
  Delete,
  UnauthorizedException,
  Logger,
  UseInterceptors,
  UploadedFile,
} from '@nestjs/common';
import {
  AddWishlist,
  CreateUserDto,
  GetUserByRoleDto,
  VerifyUserDto,
} from './dtos/user.dto';
import { UserService } from './user.service';
import { REQUEST } from '@nestjs/core';
import { Request } from 'express';
import { Roles } from 'src/decorator/role.decorator';
import { Role } from '@prisma/client';
import { AuthGuard } from 'src/auth/auth.guard';
import { FileInterceptor } from '@nestjs/platform-express/multer';
import { AWSService } from 'src/utils/aws.utils';

@Controller('user')
export class UserController {
  private readonly logger = new Logger(UserController.name);
  aws = new AWSService();
  constructor(
    private readonly userService: UserService,
    @Inject(REQUEST) private readonly request: Request,
  ) {}

  @Roles(
    Role.ADMIN,
    Role.PROVIDER,
    Role.PRESENTER,
    Role.REGULATOR,
    Role.REVIEWER,
    Role.PROFESSIONAL,
    'PUBLIC',
  )
  @UseGuards(AuthGuard)
  @Post('/add')
  async createUser(@Body() data: CreateUserDto): Promise<any> {
    const { password, ...result } = await this.userService.createUser({
      ...data,
      createdBy: this.request['user']?.role === 'ADMIN' ? 'ADMIN' : 'HIMSELF',
      isApproved:
        this.request['user']?.role === 'ADMIN' || data.role === 'PROFESSIONAL'
          ? true
          : false,
      isActive:
        this.request['user']?.role === 'ADMIN' || data.role === 'PROFESSIONAL'
          ? true
          : false,
      createdByUser: this.request['user']?.username,
    });
    this.userService.generateVerification({ ...result, password });

    return {
      success: true,
      message: 'user created successfully',
      data: result,
    };
  }

  @Roles(
    Role.ADMIN,
    Role.PROVIDER,
    Role.PRESENTER,
    Role.REGULATOR,
    Role.REVIEWER,
    Role.PROFESSIONAL,
    'PUBLIC',
  )
  @UseGuards(AuthGuard)
  @UseInterceptors(FileInterceptor('image'))
  @Put('/:id')
  async updateUser(
    @Body() data: any,
    @Param('id', ParseIntPipe) id: number,
    @UploadedFile() image: Express.Multer.File,
  ): Promise<any> {
    const user = await this.userService.getUser({ id });
    this.logger.log({
      userRole: user.role,
      authUserRole: this.request['user']?.role,
    });
    let result = {};
    if (
      user.role === 'PRESENTER' &&
      this.request['user']?.role === 'PROVIDER'
    ) {
      result = await this.userService.updateUser({
        data,
        where: { id: id },
      });
    } else if (
      user.role === 'REGULATOR' &&
      this.request['user']?.role === 'REGULATOR'
    ) {
      result = await this.userService.updateUser({
        data,
        where: { id: id },
      });
    } else if (this.request['user']?.role === 'ADMIN') {
      result = await this.userService.updateUser({
        data,
        where: { id: id },
      });
    } else if (this.request['user']?.role === 'PROFESSIONAL') {
      if (image) {
        const awsImageResponse = await this.aws.uploadFiles(image);
        result = await this.userService.updateUser({
          data: { ...data, image: awsImageResponse[0].Location },
          where: { id: id },
        });
      } else {
        result = await this.userService.updateUser({
          data: { ...data },
          where: { id: id },
        });
      }
    } else {
      result = await this.userService.updateUser({
        data: { ...data },
        where: { id: id },
      });
    }
    return {
      success: true,
      message: 'user updated successfully',
      data: result,
    };
  }

  @Roles(
    Role.ADMIN,

    Role.PRESENTER,
    Role.REGULATOR,
    Role.REVIEWER,
    Role.PROFESSIONAL,
  )
  @UseGuards(AuthGuard)

  async getUserByRole(@Body() data: GetUserByRoleDto) {

      role,
      ...(this.request['user']?.role != 'ADMIN'
        ? { createdByUser: this.request['user']?.username }

    });

      success: true,
      message: 'user fetched successfully',
      data: result,
    };
