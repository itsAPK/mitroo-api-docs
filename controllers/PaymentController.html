import { Body, Controller, Post, UseGuards } from '@nestjs/common';
import { PaymentService } from './payment.service';
import { Roles } from 'src/decorator/role.decorator';
import { AuthGuard } from 'src/auth/auth.guard';
import { Role } from '@prisma/client';

enum dashboardType {
  'DAILY',
  'MONTHLY',
  'YEARLY',
}
enum RoleType {
  'PROFESSIONAL',
  'REGULATOR',
}
@Controller('payment')
export class PaymentController {
  constructor(private readonly paymentService: PaymentService) {}
  @Roles(Role.PROFESSIONAL, Role.PROVIDER)
  @UseGuards(AuthGuard)
  @Post()
  createSession(@Body() data: any) {
    return this.paymentService.createSession(data);
  }

  @Post('webhook')
  webhook(@Body() data: any) {
    return this.paymentService.handleWebhook(data);
  }

  @Post('payout')
  createPayoutSession(@Body() data: any) {
    return this.paymentService.createPayoutSession(data);
  }

  @Post('createAccount')
  createUserAccount(@Body() data: any) {
    return this.paymentService.createPaymentUser(data);
  }

  @Post('dashboard')
  @Roles(Role.ADMIN, Role.REGULATOR, Role.PROVIDER, Role.PROFESSIONAL)
  @UseGuards(AuthGuard)
  paymentDashboard(
    @Body()
    data: {
      to: string;
      from: string;
      type: dashboardType;
      role: RoleType;
    },
  ) {
    return this.paymentService.paymentDashboard(data);
  }
  @Post('query')
  @Roles(Role.ADMIN, Role.REGULATOR, Role.PROVIDER, Role.PROFESSIONAL)
  @UseGuards(AuthGuard)
  paymentDetailsQuery(@Body() data: any) {
    return this.paymentService.getPaymentDetailsQuery(data);
  }
}
