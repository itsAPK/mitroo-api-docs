import {
  Controller,
  Post,
  Body,
  UseInterceptors,
  UploadedFiles,
  ParseFilePipe,
  UseGuards,
  ValidationPipe,
  Logger,
  Get,
  ParseIntPipe,
  Param,
  Delete,
  Put,
} from '@nestjs/common';
import { FileFieldsInterceptor } from '@nestjs/platform-express';
import { Role } from '@prisma/client';
import { AuthGuard } from 'src/auth/auth.guard';
import { Roles } from 'src/decorator/role.decorator';
import { CourseService } from './course.service';
import {
  AddCourseDto,
  CreateReviewerDetailDto,
  CreateScheduleDto,
  UpdateCourseDto,
  UpdateReviewerDetailDto,
  UpdateScheduleDto,
} from './dtos/course.dto';

@Controller('course')
export class CourseController {
  private readonly logger = new Logger(CourseController.name);

  constructor(private readonly courseService: CourseService) {}
  @Roles(Role.ADMIN, Role.PROVIDER)
  @UseGuards(AuthGuard)
  @Post('add')
  @UseInterceptors(
    FileFieldsInterceptor([
      { name: 'image', maxCount: 1 },
      { name: 'document', maxCount: 5 },
    ]),
  )
  async addCourse(
    @UploadedFiles(
      new ParseFilePipe({
        validators: [
          // new MaxFileSizeValidator({ maxSize: 1024 * 4 }),          ,
        ],
      }),
    )
    files: {
      image: Express.Multer.File;
      document: Express.Multer.File[];
    },
    @Body(new ValidationPipe()) data: AddCourseDto,
  ) {
    this.logger.log({ data });
    const { image, document } = files;
    console.log({ image, document });
    return this.courseService.createCourse(data, image, document);
  }

  @Roles(
    Role.ADMIN,
    Role.PROVIDER,
    Role.REGULATOR,
    Role.REVIEWER,
    Role.PRESENTER,
    Role.REVIEWER,
    Role.PROFESSIONAL,
  )
  @UseGuards(AuthGuard)
  @Get()
  getAllCourse(): any {
    return this.courseService.getAllCourse();
  }

  @Roles(
    Role.ADMIN,
    Role.PROVIDER,
    Role.REGULATOR,
    Role.REVIEWER,
    Role.PRESENTER,
    Role.REVIEWER,
    Role.PROFESSIONAL,
    'PUBLIC',
  )
  @UseGuards(AuthGuard)
  @Get(':id')
  getCourse(@Param('id', ParseIntPipe) id: number): any {
    return this.courseService.getCourse(id);
  }

  @Roles(Role.ADMIN, Role.PROVIDER)
  @UseGuards(AuthGuard)
  @Delete(':id')
  deleteCourse(@Param('id', ParseIntPipe) id: number): any {
    return this.courseService.deleteCourse(id);
  }

  @Roles(Role.ADMIN, Role.PROVIDER)
  @UseGuards(AuthGuard)
  @Delete(':courseId/document/:id')
  deleteCourseDocument(
    @Param('courseId', ParseIntPipe) courseId: number,
    @Param('id', ParseIntPipe) id: number,
  ): any {
    return this.courseService.deleteCourseDocument(id, courseId);
  }

  @Roles(Role.ADMIN, Role.PROVIDER, Role.REVIEWER, Role.REGULATOR)
  @UseGuards(AuthGuard)
  @Put(':id')
  updateCourse(
    @Param('id', ParseIntPipe) id: number,
    @Body() data: UpdateCourseDto,
  ): any {
    return this.courseService.updateCourse(id, data);
  }

  @Roles(
    Role.ADMIN,
    Role.PROVIDER,
    Role.REGULATOR,
    Role.REVIEWER,
    Role.PRESENTER,
    Role.REVIEWER,
    Role.PROFESSIONAL,
    'PUBLIC',
  )
  @UseGuards(AuthGuard)
  @Post('schedule')
  createSchedule(@Body() data: CreateScheduleDto): any {
    return this.courseService.createSchedule(data);
  }

  @Roles(
    Role.ADMIN,
    Role.PROVIDER,
    Role.REGULATOR,
    Role.REVIEWER,
    Role.PRESENTER,
    Role.REVIEWER,
    Role.PROFESSIONAL,
    'PUBLIC',
  )
  @UseGuards(AuthGuard)
  @Get(':id/schedule')
  getScheduleByCourseId(@Param('id', ParseIntPipe) id: number): any {
    this.logger.log({ id, fun: 'getScheduleBYCourseId' });
    return this.courseService.getScheduleByCourseId(id);
  }

  @Roles(Role.ADMIN, Role.PROVIDER)
  @UseGuards(AuthGuard)
  @Put('schedule/:id')
  updateSchedule(
    @Param('id', ParseIntPipe) id: number,
    @Body() data: UpdateScheduleDto,
  ): any {
    this.logger.log({ id, function: 'deleteSchedule' });
    return this.courseService.updateSchedule(id, data);
  }
  @Roles(Role.ADMIN, Role.PROVIDER)
  @UseGuards(AuthGuard)
  @Delete('schedule/:id')
  deleteSchedule(@Param('id', ParseIntPipe) id: number): any {
    this.logger.log({ id, function: 'deleteSchedule' });
    return this.courseService.deleteSchedule(id);
  }

  @Roles(
    Role.ADMIN,
    Role.PROVIDER,
    Role.PRESENTER,
    Role.REGULATOR,
    Role.REVIEWER,
    Role.PROFESSIONAL,
    'PUBLIC',
  )
  @UseGuards(AuthGuard)
  @Post('query')
  getCourseByQuery(@Body() data: any): any {
    this.logger.log({ function: 'getCourseByQuery' });
    return this.courseService.getCourseByQuery(data);
  }

  @Post('query/count')
  getCourseCountByQuery(@Body() data: any): any {
    this.logger.log({ function: 'getCourseByQuery' });
    return this.courseService.getCourseCountByQuery(data);
  }

  @Roles(
    Role.ADMIN,
    Role.PROVIDER,
    Role.PRESENTER,
    Role.REGULATOR,
    Role.REVIEWER,
    Role.PROFESSIONAL,
  )
  @UseGuards(AuthGuard)
  @Post('reviewer/details')
  addReviewerDetails(@Body() data: CreateReviewerDetailDto): any {
    this.logger.log({ function: 'addReviewerDetails' });
    return this.courseService.addReviewerDetails(data);
  }

  @Roles(
    Role.ADMIN,
    Role.PROVIDER,
    Role.PRESENTER,
    Role.REGULATOR,
    Role.REVIEWER,
    Role.PROFESSIONAL,
  )
  @UseGuards(AuthGuard)
  @Put('reviewer/details/:id')
  updateReviewerDetails(
    @Body() data: any,
    @Param('id', ParseIntPipe) id: number,
  ): any {
    this.logger.log({ function: 'updateReviewerDetails' });
    return this.courseService.updateReviewerDetails(data, id);
  }

  @Roles(
    Role.ADMIN,
    Role.PROVIDER,
    Role.PRESENTER,
    Role.REGULATOR,
    Role.REVIEWER,
    Role.PROFESSIONAL,
  )
  @UseGuards(AuthGuard)
  @Post('reviewer/details/query')
  getReviewerDetails(@Body() data: any): any {
    this.logger.log({ function: 'getReviewerDetails' });
    return this.courseService.getCourseByQuery(data);
  }
}
