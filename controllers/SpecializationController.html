import {
  Body,
  Controller,
  Delete,
  Get,
  Param,
  ParseIntPipe,
  Post,
  Put,
  UseGuards,
} from '@nestjs/common';
import {
  CreateLicenseDto,
  CreatePolicyDto,
  CreateSpecilizationDto,
} from './dtos/specilization.dto';
import { SpecializationService } from './specialization.service';
import { AuthGuard } from 'src/auth/auth.guard';
import { Roles } from 'src/decorator/role.decorator';
import { Role } from '@prisma/client';

@Controller('specializationPolicy')
@Roles(Role.ADMIN, Role.REGULATOR, Role.PROVIDER, Role.PROFESSIONAL)
@UseGuards(AuthGuard)
export class SpecializationController {
  constructor(private readonly specilizationService: SpecializationService) {}

  @Get('license')
  getLicense(): any {
    return this.specilizationService.getLicense();
  }

  @Post('license')
  addLicense(@Body() data: CreateLicenseDto) {
    return this.specilizationService.addLicense(data);
  }

  @Get('license/all')
  getAllLicenseType() {
    return this.specilizationService.getAllLicenseType();
  }
  @Delete('license/:id')
  deleteLicense(@Param('id', ParseIntPipe) id: number) {
    return this.specilizationService.deleteLicense(id);
  }

  @Get('specialization')
  getSpecialization(): any {
    return this.specilizationService.getSpecialization();
  }

  @Post('specialization')
  addSpecalization(@Body() data: CreateSpecilizationDto) {
    return this.specilizationService.addSpecilization(data);
  }

  @Delete('specialization/:id')
  deleteSpecialization(@Param('id', ParseIntPipe) id: number) {
    return this.specilizationService.deleteSpecilization(id);
  }

  @Post('policy')
  addPolicy(@Body() data: any) {
    return this.specilizationService.addPolicy(data);
  }

  @Post('policy/query')
  getPolicyByQuery(@Body() data: any) {
    return this.specilizationService.getPolicyByQuery(data);
  }

  @Get('policy/:id')
  getPolicy(@Param('id', ParseIntPipe) id: number) {
    return this.specilizationService.getPolicy(id);
  }

  @Get('policy')
  getAllPolicy() {
    return this.specilizationService.getAllPolicy();
  }

  @Delete('policy/:id')
  deletePolicy(@Param('id', ParseIntPipe) id: number) {
    return this.specilizationService.deletePolicy(id);
  }

  @Put('policy/:id')
  updatePolicy(@Param('id', ParseIntPipe) id: number, @Body() data: any) {
    return this.specilizationService.updatePolicy(id, data);
  }

  @Put('policyCourse/:id')
  updatePolicyCourse(@Param('id', ParseIntPipe) id: number, @Body() data: any) {
    return this.specilizationService.updatePolicyCourse(id, data);
  }

  @Delete('policyCourse/:id')
  deletePolicyCourse(@Param('id', ParseIntPipe) id: number) {
    return this.specilizationService.deletePolicyCourse(id);
  }
  @Post('policyCourse')
  addPolicyCourse(@Body() data: any) {
    return this.specilizationService.addPolicyCourse(data);
  }

  @Get('user')
  getUserSpecilization(@Body() data: any) {
    return this.specilizationService.getUserSpecilization();
  }
}
