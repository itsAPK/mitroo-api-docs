import * as bcrypt from 'bcrypt';
import { getPrismaMessage } from 'src/exception/exception.status';
import { JwtService } from '@nestjs/jwt';
import * as moment from 'moment';
import { REQUEST } from '@nestjs/core';
import { Request } from 'express';

@Injectable()

  saltOrRounds = 10;
  private readonly logger = new Logger(UserService.name);

  constructor(
    private prisma: PrismaService,
    private jwtService: JwtService,
    @Inject(REQUEST) private readonly request: Request,
  ) {}

  async createUser(data: Prisma.UserCreateInput): Promise<User> {
    try {
      const { password, ...userData } = data;
      const hash = await bcrypt.hash(password, this.saltOrRounds);
      const user = await this.prisma.user.create({
        data: { ...userData, password: hash },
      });
      return user;
    } catch (err) {
      if (err instanceof Prisma.PrismaClientKnownRequestError) {
        this.logger.error({ err: err.code || 'Prisma Exception' });
        throw new HttpException(
          getPrismaMessage(err.code, err.meta),
          HttpStatus.BAD_REQUEST,
        );
      }
      throw new HttpException(err.message, HttpStatus.BAD_REQUEST);
    }
  }

  async updateUser(params: {
    data: any;
    where: Prisma.UserWhereUniqueInput;
  }): Promise<any> {
    try {
      const { where, data } = params;
      const { password, ...user } = await this.prisma.user.update({
        data,
        where,
      });
      if (this.request['user']?.role === 'ADMIN' && 'isApproved' in data) {
        if (data.isApproved) {
          const data = {
            userId: user.id,
            token: randomBytes(16).toString('hex'),
            timestamp: moment().unix(),
          };
          const token = Buffer.from(JSON.stringify(data)).toString('base64');
          const { BASE_URL } = process.env;
          const verificationLink = `${BASE_URL}/verify?token=${token}`;

          await this.prisma.userVerification.create({
            data: {
              token: data.token,
              userId: user.id,
            },
          });
          const $ = cheerio.load(CreatedByHimselfApproved);
          const linkElement = $('#verification-link');
          // $('#email').text(user.email);
          linkElement.attr('href', verificationLink);
          sendEmail({
            to: [user.email, 'sunnyshankar081@gmail.com'],
            subject: 'Mitroo - Email Verification',
            html: $.html(),
          });
        }
      }
      if (data.password) {
        const hash = await bcrypt.hash(data.password, this.saltOrRounds);
        await this.prisma.user.update({
          data: {
            password: hash,
          },
          where,
        });
      }

      return user;
    } catch (err) {
      if (err instanceof Prisma.PrismaClientKnownRequestError) {
        this.logger.error({ err: err.code || 'Prisma Exception' });
        throw new HttpException(
          getPrismaMessage(err.code, err.meta),
          HttpStatus.BAD_REQUEST,
        );
      }
      throw new HttpException(err.message, HttpStatus.BAD_REQUEST);
    }
  }

  getUsers(

  async verifyUser(data: {
    username: string;
    email: string;
    password: string;
    role: Role;
  }): Promise<User | null> {
    this.logger.log({ data });
    const { role, username, email } = data;
    const query = username ? { username } : { email };
    const user = await this.getUser({ ...query, role });
    if (user) {
      const verified = await bcrypt.compare(data.password, user.password);
      return verified ? user : null;
    }
    return null;
  }

  async generateToken(data: any): Promise<{
    success: boolean;
    message: string;
    data: any;
  }> {
    try {
      this.logger.log({ data });
      const user = await this.verifyUser(data);
      if (!user) {
        throw new UnauthorizedException();
      }
      if (['REGULATOR', 'PROVIDER', 'PROFESSIONAL'].includes(user.role)) {
        if (!user.isActive) throw new UnauthorizedException('Inactive User');
        else if (!user.isVerified)
          throw new UnauthorizedException('Email Not Verified');
        else if (!user.isApproved)
          throw new UnauthorizedException('Approval is Pending');
      }
      const payload = {
        sub: user.id,
        username: user.username,
        role: user.role,
      };
      this.updateUser({
        data: {
          firstLogin: false,
        },
        where: {
          id: user.id,
        },
      });
      const accessToken = await this.jwtService.signAsync(payload);
      const expiredIn = process.env.EXPIRES_IN.replace(/[^\d.]/g, '');
      const expiryTime = moment().add(expiredIn, 'hours').unix();
      return {
        success: true,
        message: 'token generated',
        data: {
          firstName: user.firstName,
          lastName: user.lastName,
          accessToken,
          expiryTime,
          role: user.role,
          userId: user.id,
          firstLogin: user.firstLogin,
        },
      };
    } catch (err) {
      this.logger.error({ err: err.message || 'Internal Server Exception' });
      throw new HttpException(err.message, HttpStatus.BAD_REQUEST);
    }
  }

        where,
        include: {
          specilization: {
            include: {
              speialization: true,
              PolicyCourse: {
                include: {
                  course: {
                    include: {
                      CourseReviewerDetails: true,
                    },
                  },
                },
              },
              licence: true,
            },
          },
          user: true,
          license: true,
        },
      });
      for (const specialization of userSpecilization) {
        let currentCorePoints = 0;
        let currentElectivePoints = 0;

        const optionalCourse = await this.prisma.userCourse.findMany({
          where: {
            specializationPolicyId: specialization.specializationId,
            type: 'OPTIONAL',
            userId: specialization.userId,
          },
          include: {
            course: {
              include: {
                CourseReviewerDetails: true,
              },
            },
          },
        });
        let temp = {
          ...specialization.user,
          ...specialization,
          course: [],
        };
        for (const course of specialization.specilization.PolicyCourse) {
          const userCourse = await this.prisma.userCourse.findFirst({
            where: {
              courseId: course.courseId,
              userId: specialization.userId,
              specializationPolicyId: specialization.specializationId,
            },
          });
          const { specilization, ...userSpecilizationData } = specialization;
          const { PolicyCourse, ...specilizationData } = specilization;
          if (userCourse?.status === 'COMPLETED') {
            if (course.type === 'CORE') {
              currentCorePoints += 1;
            } else if (course.type === 'ELECTIVE') {
              currentElectivePoints += 1;
            }
            temp.course.push({
              ...course,
              status: userCourse?.status,
              purchased: true,
              ccu: course.course.CourseReviewerDetails[0]?.creditPoints,
            });
          } else {
            temp.course.push({
              ...course,
              status: userCourse ? userCourse.status : 'NONE',
              purchased: userCourse ? true : false,
              ccu: 0,
            });
          }
        }
        const currentOptionalPoints = optionalCourse
          .filter((item) => item.status === 'COMPLETED')
          .reduce(
            (total, item) =>
              total + item.course.CourseReviewerDetails[0]?.creditPoints,
            0,
          );
        result.push({
          ...temp,
          currentCorePoints,
          currentElectivePoints,
          currentOptionalPoints,
          optionalCourse,
        });
      }
      return {
        success: true,
        message: 'specilization fetched successfully',
        data: result,
      };
    } catch (err) {
      if (err instanceof Prisma.PrismaClientKnownRequestError) {
        this.logger.error({ err: err.code || 'Prisma Exception' });
        throw new HttpException(
          getPrismaMessage(err.code, err.meta),

        );

          course: {
            include: {

            },
          },
          user: true,
        },
      });
      let userCourse = temp[0];
      if (userCourse && userCourse.status === 'COMPLETED') {
        let certificate = await this.prisma.courseCertificate.findFirst({
          where: {
            userId: userCourse.userId,
            courseId: userCourse.courseId,
          },
        });
        if (!certificate)
          certificate = await this.prisma.courseCertificate.create({
            data: {
              userId: userCourse.userId,
              courseId: userCourse.courseId,
              certificateNumber: `MCR${moment().unix()}`,
              completionDate: moment(userCourse.updatedAt).format('DD-MM-YYYY'),
            },
          });
        let presenter;
        if (userCourse.course.presenterId)
          presenter = await this.prisma.user.findFirst({
            where: {
              id: userCourse.course.presenterId,
            },
          });
        return {
          success: true,
          message: 'course certificated fetched successfully',
          data: {
            firstName: userCourse.user.firstName,
            lastName: userCourse.user.lastName,
            email: userCourse.user.email,
            certificateNumber: certificate.certificateNumber,
            completionDate: certificate.completionDate,
            course: userCourse.course,
            courseUid: userCourse.course.uid,
            providerName: `${userCourse.course.createdBy.firstName} ${userCourse.course.createdBy.lastName}`,
            presenterName: presenter
              ? `${presenter.firstName} ${presenter.lastName}`
              : '',
            creditPoints:
              userCourse.course.CourseReviewerDetails[0]?.creditPoints,
          },
        };
      }
      return {
        success: false,
        message: 'please complete course to generate certificate',
        data: {},
      };
    } catch (err) {
      if (err instanceof Prisma.PrismaClientKnownRequestError) {
        this.logger.error({ err: err.code || 'Prisma Exception' });

          getPrismaMessage(err.code, err.meta),

    }
