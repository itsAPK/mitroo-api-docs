import {
  HttpException,
  HttpStatus,
  Inject,
  Injectable,
  Logger,
} from '@nestjs/common';
import { REQUEST } from '@nestjs/core';
import { Course, Prisma } from '@prisma/client';
import { getPrismaMessage } from 'src/exception/exception.status';
import { PrismaService } from 'src/prisma.service';
import { AWSService } from 'src/utils/aws.utils';
import { Request } from 'express';
import * as moment from 'moment';
import {
  CreateReviewerDetailDto,
  UpdateReviewerDetailDto,
} from './dtos/course.dto';

@Injectable()

  aws = new AWSService();

  constructor(
    private readonly prisma: PrismaService,
    @Inject(REQUEST) private readonly request: Request,
  ) {}

  async createCourse(data: any, images: any, documents: any[]): Promise<any> {
    try {
      this.logger.log({ data, user: this.request['user'] });
      const awsDocumentResponse = await this.aws.uploadFiles(documents);
      const awsImageResponse = await this.aws.uploadFiles(images);
      const uid = moment().unix().toString();
      const course = await this.prisma.$transaction(async (tx) => {
        const course = await tx.course.create({
          data: {
            ...data,
            uid: `MIT${uid}`,
            image: awsImageResponse[0].Location,
            userId: this.request['user']?.sub,
          },
        });
        const courseDocument = [];
        for (const document of awsDocumentResponse) {
          courseDocument.push({
            courseId: course.id,
            url: document.Location,
          });
        }
        await tx.courseDocument.createMany({ data: courseDocument });
        return tx.course.findUnique({
          where: { id: course.id },
          include: {
            CourseDocument: true,
          },
        });
      });

      return {
        success: true,
        message: 'course added successfully',
        data: course,
      };
    } catch (err) {
      if (err instanceof Prisma.PrismaClientKnownRequestError) {
        this.logger.error({ err: err.message || 'Prisma Exception' });
        throw new HttpException(
          getPrismaMessage(err.code, err.meta),
          HttpStatus.BAD_REQUEST,
        );
      }
      this.logger.error({ err: err.message || 'Internal Server Exception' });
      throw new HttpException(err.message, HttpStatus.BAD_REQUEST);
    }
  }

  async getAllCourse(): Promise<any> {
    try {
      const courses = await this.prisma.course.findMany({
        where: {
          userId: this.request['user']?.sub,
        },
        include: {
          CourseDocument: true,
        },
      });
      const result = [];
      for (const course of courses) {
        const isWhiteListed = await this.prisma.userWishList.findFirst({
          where: {
            courseId: course.id,
          },
        });
        result.push({ ...course, whitelisted: isWhiteListed ? true : false });
      }
      return {
        success: true,
        message: 'course fetched successfully',
        data: result,
      };
    } catch (err) {
      if (err instanceof Prisma.PrismaClientKnownRequestError) {
        this.logger.error({ err: err.message || 'Prisma Exception' });
        throw new HttpException(
          getPrismaMessage(err.code, err.meta),
          HttpStatus.BAD_REQUEST,
        );
      }
      this.logger.error({ err: err.message || 'Internal Server Exception' });
      throw new HttpException(err.message, HttpStatus.BAD_REQUEST);
    }
  }
  async addToLog(courses: Course) {
    const searchLog = await this.prisma.searchLog.findFirst({
      where: {
        courseId: courses.id,
      },
    });
    if (searchLog)
      await this.prisma.searchLog.update({
        data: { count: searchLog.count + 1 },
        where: { courseId: courses.id, id: searchLog.id },
      });
    else
      await this.prisma.searchLog.create({
        data: {
          courseId: courses.id,
          count: 1,
        },
      });
  }

  async getCourse(id: number): Promise<any> {
    try {
      const courses = await this.prisma.course.findUniqueOrThrow({
        where: { id },
        include: {
          CourseDocument: true,
          profession: true,
          category: true,
          CourseReview: true,
          CourseReviewerDetails: true,
        },
      });
      if (!this.request['user']) {
        const { CourseDocument, ...result } = courses;
        return {
          success: true,
          message: 'course fetched successfully',
          data: result,
        };
      }
      if (this.request['user']?.role === 'PROFESSIONAL' && courses) {
        this.addToLog(courses);
        const userCourse = await this.prisma.userCourse.findFirst({
          where: {
            userId: this.request['user']?.sub,
            courseId: courses.id,
          },
        });
        if (!userCourse) {
          const { CourseDocument, ...result } = courses;
          return {
            success: true,
            message: 'course fetched successfully',
            data: result,
          };
        }
      }

      return {
        success: true,
        message: 'course fetched successfully',
        data: courses,
      };
    } catch (err) {
      if (err instanceof Prisma.PrismaClientKnownRequestError) {
        this.logger.error({ err: err.message || 'Prisma Exception' });
        throw new HttpException(
          getPrismaMessage(err.code, err.meta),
          HttpStatus.BAD_REQUEST,
        );
      }
      this.logger.error({ err: err.message || 'Internal Server Exception' });
      throw new HttpException(err.message, HttpStatus.BAD_REQUEST);
    }
  }
