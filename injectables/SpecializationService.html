import {
  HttpException,
  HttpStatus,
  Inject,
  Injectable,
  Logger,
} from '@nestjs/common';
import {
  CreateLicenseDto,
  CreatePolicyDto,
  CreateSpecilizationDto,
} from './dtos/specilization.dto';
import { PrismaService } from 'src/prisma.service';
import { REQUEST } from '@nestjs/core';
import { Prisma, Specialization } from '@prisma/client';
import { getPrismaMessage } from 'src/exception/exception.status';
import * as moment from 'moment';
import { log } from 'console';

@Injectable()
export class SpecializationService {
  private readonly logger = new Logger(SpecializationService.name);

  constructor(
    private readonly prisma: PrismaService,
    @Inject(REQUEST) private readonly request: Request,

  async addLicense(data: CreateLicenseDto): Promise<any> {
    try {
      const license = await this.prisma.license.create({
        data: { ...data, userId: this.request['user']?.sub },
      });
      return {
        success: true,
        message: 'license created successfully',
        data: license,
      };
    } catch (err) {
      if (err instanceof Prisma.PrismaClientKnownRequestError) {
        this.logger.error({ err: err.message || 'Prisma Exception' });
        throw new HttpException(
          getPrismaMessage(err.code, err.meta),
          HttpStatus.BAD_REQUEST,
        );
      }
      this.logger.error({ err: err.message || 'Internal Server Exception' });
      throw new HttpException(err.message, HttpStatus.BAD_REQUEST);
    }
  }

  async addPolicy(data: any): Promise<any> {
    try {
      const { policyCourses, expiry, ...policyData } = data;
      log({ policyData, policyCourses });
      const specializationPolicy = await this.prisma.$transaction(
        async (tx) => {
          const specialization = await tx.specializationPolicy.create({
            data: {
              ...policyData,
              validTill: moment().add({
                months: expiry.month,
                years: expiry.year,
              }),
              policyNumber: `PL${moment().unix()}`,
              userId: this.request['user']?.sub,
            },
          });
          const courses = policyCourses.map((course) => ({
            courseId: course.courseId,
            type: course.type,
            specializationPolicyId: specialization.id,
          }));
          await tx.policyCourse.createMany({ data: courses });
          return tx.specializationPolicy.findUnique({
            where: { id: specialization.id },
            include: {
              PolicyCourse: true,
            },
          });
        },
      );

      return {
        success: true,
        message: 'specialization added successfully',
        data: specializationPolicy,
      };
    } catch (err) {
      if (err instanceof Prisma.PrismaClientKnownRequestError) {
        this.logger.error({ err: err.message || 'Prisma Exception' });
        throw new HttpException(
          getPrismaMessage(err.code, err.meta),
          HttpStatus.BAD_REQUEST,
        );
      }
      this.logger.error({ err: err.message || 'Internal Server Exception' });
      throw new HttpException(err.message, HttpStatus.BAD_REQUEST);
    }
  }

  async deletePolicy(id: number): Promise<any> {
    try {

        async (tx) => {
          await tx.policyCourse.deleteMany({

          });
          const specialization = await tx.specializationPolicy.delete({
            where: { id },
          });

      return {
        success: true,
        message: 'specialization deleted successfully',
        data: specializationPolicy,

  async getUserSpecilization(): Promise<any> {
    try {
      const userLicense = await this.prisma.userLicense.findMany({
        where: {
          userId: this.request['user'].sub,
        },
      });
      const regulatorId = userLicense
        .filter((license) => license.regulatorId !== null)
        .map((license) => license.regulatorId);

      const specialization = await this.prisma.specializationPolicy.findMany({
        where: {
          userId: {
            in: regulatorId,
          },
        },
        include: {
          PolicyCourse: {
            include: {
              course: true,
            },
          },
        },
      });
      return {
        success: true,
        message: 'specialization course fetched successfully',
        data: specialization,
      };
    } catch (err) {
      if (err instanceof Prisma.PrismaClientKnownRequestError) {
        this.logger.error({ err: err.message || 'Prisma Exception' });
        throw new HttpException(
          getPrismaMessage(err.code, err.meta),
          HttpStatus.BAD_REQUEST,
        );
      }
      this.logger.error({ err: err.message || 'Internal Server Exception' });
      throw new HttpException(err.message, HttpStatus.BAD_REQUEST);
    }
  }

  async addToLog(specialization: Specialization) {
    const searchLog = await this.prisma.searchLog.findFirst({
      where: {
        specializationPolicyId: specialization.id,
      },
    });
    if (searchLog)
      await this.prisma.searchLog.update({
        data: { count: searchLog.count + 1 },
        where: { id: searchLog.id },
      });
    else
      await this.prisma.searchLog.create({
        data: {
          specializationPolicyId: specialization.id,
          count: 1,
        },
      });
  }
  async getPolicy(id: number): Promise<any> {
    try {
      const policy = await this.prisma.specializationPolicy.findUnique({
        where: { id },
        include: {
          PolicyCourse: {
            include: {
              course: {
                include: {
                  CourseReviewerDetails: true,
                },
              },
            },
          },
          licence: true,
          speialization: true,
        },
      });
      if (this.request['user']?.role === 'PROFESSIONAL' && policy)
        this.addToLog(policy);
      return {
        success: true,
        message: 'specialization policy fetched successfully',
        data: policy,

    } catch (err) {
      if (err instanceof Prisma.PrismaClientKnownRequestError) {

  async updatePolicy(id: number, data: any): Promise<any> {

          data: {
