import { HttpException, HttpStatus, Inject, Injectable } from '@nestjs/common';
import { REQUEST } from '@nestjs/core';
import { Prisma } from '@prisma/client';
import { Request } from 'express';
import * as moment from 'moment';
import { getPrismaMessage } from 'src/exception/exception.status';
import { PrismaService } from 'src/prisma.service';
import Stripe from 'stripe';
import * as _ from 'lodash';
import { timestamp } from 'rxjs';

@Injectable()
export class PaymentService {
  stripe = new Stripe(process.env.STRIPE_KEY, {
    apiVersion: '2023-10-16',
  });
  constructor(
    private readonly prisma: PrismaService,
    @Inject(REQUEST) private readonly request: Request,
  ) {}

  async createSession(data: any): Promise<any> {
    try {
      if (data.paymentFor && data.paymentFor === 'USER') {
        const { regulatorId, amount, success_url, cancel_url, courseId } = data;
        const user = await this.prisma.user.findUnique({
          where: {
            id: regulatorId,
          },
        });
        const orderId = `MITROO_${moment().unix()}`;
        const session = await this.stripe.checkout.sessions.create({
          line_items: [
            {
              price_data: {
                unit_amount: amount * 100,
                currency: 'usd',
                product_data: {
                  name: `User - ${user.username}`,
                  description: `Payment for User ${user.username}`,
                },
              },
              quantity: 1,
            },
          ],
          mode: 'payment',
          // payment_method_types: ['card'],
          success_url,
          cancel_url,
          metadata: {
            orderId,
          },
        });
        await this.prisma.paymentLog.create({
          data: {
            refId: orderId,
            amount: session.amount_total,
            course: courseId,
            paymentFor: 'USER',
            regulator: regulatorId,
            userId: this.request['user']?.sub,
          },
        });
        return {
          success: true,
          message: 'session generated successfully',
          data: session,
        };
      } else {
        const { scheduleId, success_url, cancel_url } = data;
        const orderId = `MITROO_${moment().unix()}`;

        const items = [];
        for (const id of scheduleId) {
          const schedule = await this.prisma.courseSchedule.findUnique({
            where: {
              id,
            },
            include: {
              course: true,
            },
          });
          if (schedule) {
            await this.prisma.paymentLog.create({
              data: {
                refId: orderId,
                amount: schedule.course.price,
                courseSchedule: [id],
                userId: this.request['user']?.sub,
                course: schedule.course.id,
              },
            });
            items.push({
              price_data: {
                unit_amount: schedule.course.price * 100,
                currency: 'usd',
                product_data: {
                  name: schedule.course.name,
                  description: schedule.course.description,
                  images: [schedule.course.image],
                },
              },
              quantity: 1,
            });
          }
        }
        const session = await this.stripe.checkout.sessions.create({
          line_items: items,
          mode: 'payment',
          // payment_method_types: ['card'],
          success_url,
          cancel_url,
          metadata: {
            orderId,
          },
        });
        return {
          success: true,
          message: 'session generated successfully',
          data: session,
        };
      }
    } catch (err) {
      if (err instanceof Prisma.PrismaClientKnownRequestError) {
        throw new HttpException(
          getPrismaMessage(err.code, err.meta),
          HttpStatus.BAD_REQUEST,
        );
      }
      throw new HttpException(err.message, HttpStatus.BAD_REQUEST);
    }
  }

  async groupDataByTime(data, timeUnit) {
    return _.chain(data)
      .groupBy((item) => {
        const roundedTimestamp = moment(item.timestamp);

        switch (timeUnit) {
          case 'day':
            return roundedTimestamp.format('YYYY-MM-DD');
          case 'week':
            return roundedTimestamp.startOf('week').format('YYYY-MM-DD');
          case 'month':
            return roundedTimestamp.startOf('month').format('YYYY-MM-DD');
          default:
            return null;
        }
      })
      .map((groupedData, timestamp) => ({
        timestamp,
        total: groupedData.reduce((acc, obj) => acc + obj.buyedCourse.price, 0),
        name: moment(timestamp).format(
          timeUnit === 'week' ? 'MMMM' : timeUnit === 'month' ? 'YYYY' : 'dddd',
        ),
        data: groupedData,
      }))
      .value();
  }

  async paymentDashboard(data: any): Promise<any> {
    try {
      if (this.request['user']?.role != 'ADMIN') {
        const paymentObjs = await this.prisma.paymentLog.findMany({
          where: {
            buyedCourse: {
              userId: this.request['user']?.sub,
            },
            createdAt: {
              gte: moment(data.from, 'YYYY-MM-DD').toDate(),
              lte: moment(data.to, 'YYYY-MM-DD').toDate(),
            },
          },
          include: {
            user: true,
            buyedCourse: {
              include: {
                createdBy: true,
              },
            },
          },
        });

        if (paymentObjs.length > 0) {
          const result = await this.groupDataByTime(paymentObjs, data.type);
          console.log({ result });
          return {
            success: true,
            message: 'api executed successfully',
            data: result,
          };
        }
      } else if (this.request['user']?.role === 'ADMIN') {
        const paymentObjs = await this.prisma.paymentLog.findMany({
          where: {
            paymentFor: data.role === 'PROFESSIONAL' ? 'COURSE' : 'USER',
            createdAt: {
              gte: data.from
                ? moment(data.from, 'YYYY-MM-DD').toDate()
                : moment().subtract(30, 'days').toDate(),
              lte: data.to
                ? moment(data.to, 'YYYY-MM-DD').toDate()
                : moment().toDate(),
            },
          },
          include: {
            user: true,
            buyedCourse: {
              include: {
                createdBy: true,
              },
            },
          },
        });
        if (paymentObjs.length > 0) {
          const result = await this.groupDataByTime(paymentObjs, data.type);
          console.log({ result });
          return {
            success: true,
            message: 'api executed successfully',
            data: result,
          };
        }
      }
      return {
        success: true,
        message: 'no data found',
        data: {},
      };
    } catch (err) {
      console.log({ err });
      if (err instanceof Prisma.PrismaClientKnownRequestError) {
        throw new HttpException(
          getPrismaMessage(err.code, err.meta),
          HttpStatus.BAD_REQUEST,
        );
      }
      throw new HttpException(err.message, HttpStatus.BAD_REQUEST);
    }
  }

  async handleWebhook(data: any) {
    try {
      if (data.type === 'checkout.session.completed') {
        const object = data.data.object;
        const orderId = object.metadata.orderId;
        console.log({ orderId });
        await this.prisma.paymentLog.updateMany({
          data: {
            status: 'SUCCESS',
          },
          where: {
            refId: orderId,
          },
        });
        const paymentData = await this.prisma.paymentLog.findMany({
          where: {
            refId: orderId,
          },
        });
        if (paymentData[0].paymentFor === 'COURSE') {
          for (const obj of paymentData) {
            const schedule = await this.prisma.courseSchedule.findUnique({
              where: {
                id: obj.courseSchedule[0],
              },
              include: {
                course: true,
              },
            });
            if (schedule) {
              // await this.prisma.userCourse.create({
              //   data: {
              //     userId,
              //     courseId: schedule.course.id,
              //     courseScheduleId: schedule.id,
              //   },
              // });
              const userSpecilization =
                await this.prisma.userSpecilization.findFirst({
                  where: {
                    userId: obj.userId,
                    specilization: {
                      PolicyCourse: {
                        some: {
                          courseId: schedule.course.id,
                        },
                      },
                    },
                  },
                  include: {
                    specilization: true,
                  },
                });
              await this.prisma.userCourse.updateMany({
                data: {
                  status: 'INPROCESS',
                  isPurchased: true,
                  specializationPolicyId: userSpecilization
                    ? userSpecilization.specializationId
                    : null,
                },
                where: {
                  userId: obj.userId,
                  courseId: schedule.course.id,
                  courseScheduleId: schedule.id,
                },
              });
            }
          }
          await this.prisma.userCart.deleteMany({
            where: {
              userId: paymentData[0].userId,
            },
          });
        } else {
          for (const obj of paymentData) {
            const { course, regulator } = obj;
            await this.prisma.course.update({
              data: {
                regulatorId: regulator,
              },
              where: {
                id: course,
              },
            });
          }
        }
      }
      await this.prisma.webhookLog.create({
        data: {
          response: data,
        },
      });
      return { success: true };
    } catch (err) {
      if (err instanceof Prisma.PrismaClientKnownRequestError) {
        throw new HttpException(
          getPrismaMessage(err.code, err.meta),
          HttpStatus.BAD_REQUEST,
        );
      }
      throw new HttpException(err.message, HttpStatus.BAD_REQUEST);
    }
  }
}
