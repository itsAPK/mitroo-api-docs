import { Module } from '@nestjs/common';
import { UserModule } from './user/user.module';
import { AdminModule } from './admin/admin.module';
import { LoggerModule } from 'nestjs-pino';

import { JwtModule } from '@nestjs/jwt';
import { ProfessionModule } from './profession/profession.module';
import { CategoryModule } from './category/category.module';
import { CourseModule } from './course/course.module';
import { SpecializationModule } from './specialization/specialization.module';
import { SearchModule } from './search/search.module';
import { CartModule } from './cart/cart.module';
import { PaymentModule } from './payment/payment.module';
import { PrometheusModule } from "@willsoto/nestjs-prometheus";
import { LokiLoggerModule } from 'nestjs-loki-logger';
@Module({
  imports: [
    LoggerModule.forRoot({
      pinoHttp: {
        transport: {
          target: 'pino-pretty',
        },
      },
    }),
    JwtModule.register({
      global: true,
      secret: process.env.SECRET_KEY,
      signOptions: { expiresIn: process.env.EXPIRES_IN },
    }),
    PrometheusModule.register(),
    LokiLoggerModule.forRoot({
      lokiUrl: 'http://192.168.1.4:3100', // loki server
      labels: {
        label: 'testing', // app level labels, these labels will be attached to every log in the application
      },
      logToConsole: true,
      gzip: false, // contentEncoding support gzip or not
    }),
    UserModule,
    AdminModule,
    ProfessionModule,
    CategoryModule,
    CourseModule,
    SpecializationModule,
    SearchModule,
    CartModule,
    PaymentModule,
  ],
  controllers: [],
  providers: [],
})
export class AppModule {}
